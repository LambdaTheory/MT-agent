# MT Agent Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a manually triggered local MT operations agent that collects Alipay dashboard 1-day, 7-day, and 30-day product data, then generates raw JSON, XLSX, Markdown, and a run log.

**Architecture:** The first version is a TypeScript CLI. The crawler uses Playwright for browser automation, extraction normalizes Ant Design table data, the analyzer produces product-level recommendations keyed by platform product ID, and the report layer writes XLSX and Markdown outputs.

**Tech Stack:** Node.js, TypeScript, Playwright, Vitest, xlsx-js-style, filesystem storage.

```ts
import { readFile } from 'node:fs/promises';
import type { AgentConfig, PeriodKey } from '../domain/types.js';

const PERIODS = new Set<PeriodKey>(['1d', '7d', '30d']);

function stringField(value: unknown, name: string): string {
  if (typeof value !== 'string' || value.trim() === '') throw new Error(`Invalid config field: ${name}`);
  return value;
}

function positiveNumber(value: unknown, name: string): number {
  if (typeof value !== 'number' || !Number.isFinite(value) || value <= 0) throw new Error(`Invalid config field: ${name}`);
  return value;
}

export function parseAgentConfig(value: unknown): AgentConfig {
  if (!value || typeof value !== 'object') throw new Error('Config must be an object');
  const record = value as Record<string, unknown>;
  if (!Array.isArray(record.periods) || record.periods.length === 0) throw new Error('Invalid config field: periods');

  return {
    targetUrl: stringField(record.targetUrl, 'targetUrl'),
    periods: record.periods.map((period) => {
      if (typeof period !== 'string' || !PERIODS.has(period as PeriodKey)) throw new Error(`Unsupported period: ${String(period)}`);
      return period as PeriodKey;
    }),
    preferredPageSize: positiveNumber(record.preferredPageSize, 'preferredPageSize'),
    outputDir: stringField(record.outputDir, 'outputDir'),
    browserProfileDir: stringField(record.browserProfileDir, 'browserProfileDir'),
  };
}

export async function loadConfig(path = 'config/agent.config.json'): Promise<AgentConfig> {
  return parseAgentConfig(JSON.parse(await readFile(path, 'utf8')));
}
```

```ts
import { normalizeText } from './normalizeText.js';

export interface ExtractedTable { headers: string[]; rows: string[][]; }
export interface ColumnDefinition { sourceIndex: number; headers: string[]; readKind: 'product' | 'spu' | 'plain'; }
type StructuredFieldKind = 'product' | 'spu';
const ACTION_HEADER_NAMES = new Set(['action', 'actions', '操作']);

export function sanitizeStructuredId(value: string, kind: StructuredFieldKind): string {
  let result = value;
  if (kind === 'product') result = result.replace(/^商品ID[:：\s-]*/i, '');
  if (kind === 'spu') result = result.replace(/^SPUID[:：\s-]*/i, '').replace(/^SPU ID[:：\s-]*/i, '');
  return normalizeText(result.replace(/^ID[:：\s-]*/i, '').replace(/复制$/i, '').replace(/copy$/i, ''));
}

export function extractStructuredNameAndId(parts: string[], kind: StructuredFieldKind): [string, string] {
  const idMatchers = kind === 'product' ? [/商品ID/i, /^id[:：]/i] : [/SPUID/i, /SPU ID/i, /^id[:：]/i];
  const filtered = parts.map(normalizeText).filter((part) => part && !/^复制$|^copy$/i.test(part));
  const idPart = filtered.find((part) => idMatchers.some((matcher) => matcher.test(part))) ?? filtered[1] ?? '';
  const namePart = filtered.find((part) => part !== idPart) ?? filtered[0] ?? '';
  return [normalizeText(namePart), sanitizeStructuredId(idPart, kind)];
}

export function getColumnDefinitionsForHeaders(headers: string[]): ColumnDefinition[] {
  return headers.flatMap((header, index) => {
    const headerText = normalizeText(header);
    const lower = headerText.toLowerCase();
    if (headerText === '' || ACTION_HEADER_NAMES.has(headerText) || ACTION_HEADER_NAMES.has(lower)) return [];
    if (headerText === '商品信息') return [{ sourceIndex: index, headers: ['商品名称', '商品ID'], readKind: 'product' }];
    if (headerText === 'SPU信息') return [{ sourceIndex: index, headers: ['SPU名称', 'SPUID'], readKind: 'spu' }];
    return [{ sourceIndex: index, headers: [headerText], readKind: 'plain' }];
  });
}
```

```ts
import type { PeriodProductMetrics, RawTableData } from '../domain/types.js';

function findHeaderIndex(headers: string[], matchers: string[]): number {
  return headers.findIndex((header) => matchers.some((matcher) => header.includes(matcher)));
}

function parseCount(value: string | undefined): number {
  const numeric = Number((value ?? '').replace(/,/g, '').trim());
  return Number.isFinite(numeric) ? numeric : 0;
}

export function normalizeRowsForPeriod(table: RawTableData): PeriodProductMetrics[] {
  const indexes = {
    productName: findHeaderIndex(table.headers, ['商品名称', '商品信息', '商品', '标题']),
    platformProductId: findHeaderIndex(table.headers, ['商品ID']),
    spuName: findHeaderIndex(table.headers, ['SPU名称', 'SPU信息']),
    spuId: findHeaderIndex(table.headers, ['SPUID']),
    visits: findHeaderIndex(table.headers, ['频道访问次数', '访问次数', '访问']),
    createdOrders: findHeaderIndex(table.headers, ['创建订单数']),
    signedOrders: findHeaderIndex(table.headers, ['签约订单数']),
    reviewedOrders: findHeaderIndex(table.headers, ['审出订单数']),
    shippedOrders: findHeaderIndex(table.headers, ['发货订单数']),
  };
  const missing = Object.entries(indexes).filter(([key, value]) => !['spuName', 'spuId'].includes(key) && value < 0).map(([key]) => key);
  if (missing.length > 0) throw new Error(`Missing required headers for ${table.period}: ${missing.join(', ')}. Actual headers: ${table.headers.join(', ')}`);
  return table.rows.map((row) => ({
    period: table.period,
    productName: row[indexes.productName] ?? '',
    platformProductId: row[indexes.platformProductId] ?? '',
    spuName: indexes.spuName >= 0 ? row[indexes.spuName] : undefined,
    spuId: indexes.spuId >= 0 ? row[indexes.spuId] : undefined,
    visits: parseCount(row[indexes.visits]),
    createdOrders: parseCount(row[indexes.createdOrders]),
    signedOrders: parseCount(row[indexes.signedOrders]),
    reviewedOrders: parseCount(row[indexes.reviewedOrders]),
    shippedOrders: parseCount(row[indexes.shippedOrders]),
  }));
}
```

```ts
import type { Level, PeriodKey, PeriodProductMetrics, ProductAnalysisRow, ProductMetrics, RecommendationAction } from '../domain/types.js';

const PERIODS: PeriodKey[] = ['1d', '7d', '30d'];
const LOW_EXPOSURE_VISITS = 100;
const HIGH_CONFIDENCE_30D_VISITS = 100;

function level(score: number): Level {
  if (score >= 70) return '高';
  if (score >= 35) return '中';
  return '低';
}

function rate(metrics: ProductMetrics | null): number {
  return metrics && metrics.visits > 0 ? metrics.shippedOrders / metrics.visits : 0;
}

function reason(metrics: Record<PeriodKey, ProductMetrics | null>, action: RecommendationAction): string {
  const p = (period: PeriodKey) => `${period}访问${metrics[period]?.visits ?? 0}，发货${metrics[period]?.shippedOrders ?? 0}`;
  const base = `${p('1d')}；${p('7d')}；${p('30d')}`;
  if (action === '疑似失活') return `${base}；30天有访问但没有发货，短期也没有恢复信号。`;
  if (action === '疑似价格问题') return `${base}；30天曾有成交信号，但近期访问仍在且发货转弱，优先检查价格和竞争力。`;
  if (action === '建议补链') return `${base}；低曝光下已有发货信号，建议增加同款链接或曝光。`;
  if (action === '高曝光低转化') return `${base}；访问较高但转化弱，检查价格、主图、标题和库存。`;
  if (action === '稳定优质') return `${base}；多个周期有稳定发货信号。`;
  if (action === '建议加曝光') return `${base}；转化表现可用，建议增加曝光。`;
  return `${base}；当前信号不够明确，继续观察。`;
}

function confidence(metrics: Record<PeriodKey, ProductMetrics | null>): Level {
  const thirty = metrics['30d'];
  if (thirty && thirty.visits >= HIGH_CONFIDENCE_30D_VISITS && metrics['7d'] && metrics['1d']) return '高';
  if (thirty && thirty.visits >= 30) return '中';
  return '低';
}

function analyzeOne(productName: string, platformProductId: string, metrics: Record<PeriodKey, ProductMetrics | null>): ProductAnalysisRow {
  const one = metrics['1d'];
  const seven = metrics['7d'];
  const thirty = metrics['30d'];
  const noRecentShipments = (one?.shippedOrders ?? 0) === 0 && (seven?.shippedOrders ?? 0) === 0;
  const lowExposureWithShipment = ((one?.visits ?? Infinity) < LOW_EXPOSURE_VISITS && (one?.shippedOrders ?? 0) > 0) || ((seven?.visits ?? Infinity) < LOW_EXPOSURE_VISITS && (seven?.shippedOrders ?? 0) > 0);
  let action: RecommendationAction = '继续观察';
  let riskScore = 10;
  let opportunityScore = 10;

  if (lowExposureWithShipment) { action = '建议补链'; opportunityScore = 85; riskScore = 15; }
  else if (thirty && thirty.visits > 0 && thirty.shippedOrders === 0 && noRecentShipments) { action = '疑似失活'; riskScore = thirty.visits >= 100 ? 85 : 65; opportunityScore = 5; }
  else if (thirty && thirty.shippedOrders > 0 && noRecentShipments && ((seven?.visits ?? 0) > 0 || (one?.visits ?? 0) > 0)) { action = '疑似价格问题'; riskScore = 55; opportunityScore = 25; }
  else if ((seven?.visits ?? 0) >= 300 && rate(seven) < 0.01) { action = '高曝光低转化'; riskScore = 70; opportunityScore = 20; }
  else if ((one?.shippedOrders ?? 0) > 0 && (seven?.shippedOrders ?? 0) > 0 && (thirty?.shippedOrders ?? 0) > 0) { action = '稳定优质'; riskScore = 5; opportunityScore = 70; }
  else if (rate(seven) > 0.02 || rate(thirty) > 0.02) { action = '建议加曝光'; riskScore = 10; opportunityScore = 60; }

  const ref = one ?? seven ?? thirty;
  return { productName, platformProductId, spuName: ref?.spuName, spuId: ref?.spuId, metrics, riskScore, opportunityScore, riskLevel: level(riskScore), opportunityLevel: level(opportunityScore), action, confidence: confidence(metrics), reason: reason(metrics, action) };
}

export function analyzeProducts(rows: PeriodProductMetrics[]): ProductAnalysisRow[] {
  const grouped = new Map<string, { productName: string; metrics: Record<PeriodKey, ProductMetrics | null> }>();
  for (const row of rows) {
    const existing = grouped.get(row.platformProductId) ?? { productName: row.productName, metrics: { '1d': null, '7d': null, '30d': null } };
    existing.metrics[row.period] = row;
    grouped.set(row.platformProductId, existing);
  }
  return Array.from(grouped.entries()).map(([id, value]) => analyzeOne(value.productName, id, value.metrics)).sort((a, b) => b.riskScore + b.opportunityScore - (a.riskScore + a.opportunityScore));
}
```

```ts
export function createRunLog(startTime: string, targetUrl: string) {

  return {
```

```ts
import * as XLSX from 'xlsx-js-style';
import type { DailyReportData, PeriodKey, ProductAnalysisRow } from '../domain/types.js';

const RAW_SHEETS: Record<PeriodKey, string> = { '1d': '1天原始数据', '7d': '7天原始数据', '30d': '30天原始数据' };
function metric(row: ProductAnalysisRow, period: PeriodKey, key: 'visits' | 'createdOrders' | 'shippedOrders') { return row.metrics[period]?.[key] ?? 0; }
function shippedRate(row: ProductAnalysisRow) { const m = row.metrics['30d']; return m && m.visits > 0 ? `${((m.shippedOrders / m.visits) * 100).toFixed(2)}%` : '0.00%'; }

export function buildWorkbook(data: DailyReportData): XLSX.WorkBook {
  const workbook = XLSX.utils.book_new();
  for (const table of data.rawTables) XLSX.utils.book_append_sheet(workbook, XLSX.utils.aoa_to_sheet([table.headers, ...table.rows]), RAW_SHEETS[table.period]);
  const analysis = [['商品名称', '平台商品ID', '1天访问', '1天创建', '1天发货', '7天访问', '7天创建', '7天发货', '30天访问', '30天创建', '30天发货', '30天发货率', '风险分', '机会分', '风险等级', '机会等级', '建议动作', '置信度', '判定原因'], ...data.analysisRows.map((row) => [row.productName, row.platformProductId, metric(row, '1d', 'visits'), metric(row, '1d', 'createdOrders'), metric(row, '1d', 'shippedOrders'), metric(row, '7d', 'visits'), metric(row, '7d', 'createdOrders'), metric(row, '7d', 'shippedOrders'), metric(row, '30d', 'visits'), metric(row, '30d', 'createdOrders'), metric(row, '30d', 'shippedOrders'), shippedRate(row), row.riskScore, row.opportunityScore, row.riskLevel, row.opportunityLevel, row.action, row.confidence, row.reason])];
  XLSX.utils.book_append_sheet(workbook, XLSX.utils.aoa_to_sheet(analysis), '商品综合分析');
  return workbook;
}

export function writeWorkbookBuffer(data: DailyReportData): Buffer {
  return Buffer.from(XLSX.write(buildWorkbook(data), { bookType: 'xlsx', type: 'array' }));
}
```

```ts
export function dedupeRowsByProductId(headers: string[], rows: string[][]): string[][] {
  const idIndex = headers.findIndex((header) => header.includes('商品ID'));
  if (idIndex < 0) return rows;
  const byId = new Map<string, string[]>();
  for (const row of rows) if (row[idIndex]) byId.set(row[idIndex], row);
  return Array.from(byId.values());
}

export function isCollectionComplete(dedupedRowCount: number, displayedTotalCount: number | null, nextPageDisabled: boolean): boolean {
  return displayedTotalCount === null ? nextPageDisabled : dedupedRowCount === displayedTotalCount;
}
```

```ts
import { mkdir, writeFile } from 'node:fs/promises';
import { loadConfig } from '../config/loadConfig.js';
import { crawlDashboard } from '../crawler/dashboardCrawler.js';
import { normalizeRowsForPeriod } from '../extractor/normalizeRows.js';
import { analyzeProducts } from '../analyzer/analyzeProducts.js';
import { buildMarkdownReport } from '../report/buildMarkdown.js';
import { writeWorkbookBuffer } from '../report/buildWorkbook.js';
import { buildOutputPaths } from '../storage/outputPaths.js';
import { createRunLog } from '../storage/runLog.js';

function today(): string {
  return new Date().toISOString().slice(0, 10);
}

async function main(): Promise<void> {
  const config = await loadConfig();
  const date = today();
  const paths = buildOutputPaths(config.outputDir, date);
  const log = createRunLog(new Date().toISOString(), config.targetUrl);
  await mkdir(paths.dir, { recursive: true });

  try {
    log.addEvent('Starting dashboard crawl');
    const rawTables = await crawlDashboard(config);
    for (const table of rawTables) {
      log.addPeriodStats(table.collection);
      await writeFile(paths.raw[table.period], JSON.stringify(table, null, 2), 'utf8');
    }

    const metrics = rawTables.flatMap(normalizeRowsForPeriod);
    const analysisRows = analyzeProducts(metrics);
    const report = { date, rawTables, analysisRows, incomplete: rawTables.some((table) => !table.collection.complete) };
    await writeFile(paths.workbook, writeWorkbookBuffer(report));
    await writeFile(paths.markdown, buildMarkdownReport(report), 'utf8');
    log.addEvent(`Wrote workbook: ${paths.workbook}`);
    log.addEvent(`Wrote markdown: ${paths.markdown}`);
  } catch (error) {
    log.addEvent(`ERROR: ${error instanceof Error ? error.message : String(error)}`);
    throw error;
  } finally {
    await writeFile(paths.log, log.toText(), 'utf8');
  }
}

main().catch((error) => {
  console.error(error instanceof Error ? error.message : error);
  process.exitCode = 1;
});
```
